For Loops
"""

print(range(10))

#Using loops to print 1 through 10
for i in range(10):
    print(i)

#Using loops with lists
num_ls = [0, 3, 9, 91]

for i in range(len(num_ls)):
    print(num_ls[i])

#Lets say we want to add 1 to each item in num_ls
for i in range(len(num_ls)):
    num_ls[i] += 1
    #num_ls[i] = num_ls[i] + 1 #same as +=
    
print(num_ls)
