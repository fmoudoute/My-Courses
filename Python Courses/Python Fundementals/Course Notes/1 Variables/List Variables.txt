"""
Basic List Manipulation
"""

num_ls = [17,23,43,24]

#Finding an item from a list
num_ls[2]

#Length of a list
len(num_ls)

#Pop an item base on location
num_ls.pop(3)
print(num_ls)

#Remove an item base on name
num_ls.remove(17)
print(num_ls)

#Add an item to a list
num_ls.append(12)
print(num_ls)

#Sort a list
num_ls.sort()
print(num_ls)
